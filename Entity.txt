CREATE TABLE IF NOT EXISTS `board20240115`.`user` (
  `email` VARCHAR(100) NOT NULL COMMENT '사용자 이메일',
  `password` VARCHAR(255) NOT NULL COMMENT '사용자의 로그인 비밀번호(암호화 되어 있음)',
  `nickname` VARCHAR(45) NOT NULL COMMENT '사용자의 닉네임',
  `tel_number` VARCHAR(13) NOT NULL COMMENT '사용자의 휴대전화 번호',
  `address` TEXT NOT NULL COMMENT '사용자의 주소',
  `address_datail` TEXT NOT NULL,
  `profile_image` TEXT NULL COMMENT '사용자의 프로필 사진 URL',
  PRIMARY KEY (`email`),
  UNIQUE INDEX `email_UNIQUE` (`email` ASC) VISIBLE,
  UNIQUE INDEX `tel_number_UNIQUE` (`tel_number` ASC) VISIBLE)
ENGINE = InnoDB
COMMENT = '게시물 프로젝트의 사용자 테이블'
